/**
 * @defgroup pkg_picol Picol
 * @ingroup  pkg
 * @brief    Picol is a Tcl-alike interpreter in 5000 lines of code.
 * @see      https://chiselapp.com/user/dbohdan/repository/picol/index
 * @see      http://oldblog.antirez.com/post/picol.html
 * @see      https://wiki.tcl-lang.org/page/Picol
 *
 * The contrib part supplies:
 * - a shell loop `riot_picol_shell_run`
 * - a command `exec` which allow to call a command registered for the standard Shell
 *
 * Typical head with custom config:
 * ```
 *    #define PICOL_CONFIGURATION
 *    #define PICOL_MAX_STR           256
 *    #define PICOL_EVAL_BUF_SIZE     512
 *    #define PICOL_SOURCE_BUF_SIZE     1
 *    #define PICOL_MAX_LEVEL          10
 *    /* A minimal Picol configuration. */
 *    #define PICOL_FEATURE_ARRAYS  0
 *    #define PICOL_FEATURE_GLOB    0
 *    #define PICOL_FEATURE_INTERP  0
 *    #define PICOL_FEATURE_IO      0
 *    #define PICOL_FEATURE_PUTS    1
 *    
 *    #include "picol_shell.h"
 *    
 *    #define PICOL_IMPLEMENTATION
 *    #include "picol.h"
 * ```
 *
 * Typical use:
 * ```
 *    struct picolInterp interp;
 *    picolInitInterp(&interp);
 *    picolRegisterCoreCommands(&interp);
 *    (void)picolRegisterCommand(&interp, "exec", riot_picol_command_exec, my_commands);
 *    (void)picolSetVar(&interp, "RIOT_BOARD", RIOT_BOARD);
 *    picolEval(&interp, "puts \"This is Picol shell on $RIOT_BOARD\"");
 *    riot_picol_shell_run(&interp);
 * ```
 */
